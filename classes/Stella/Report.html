    def process
      self.class.plugin_order.each do |name|
        klass = self.class.plugins[name]
        Stella.ld "processing #{name}"
        plugin = klass.new self
        plugin.process(filter)
        self.send("#{name}=", plugin)
      end
      @processed = true
    end
