    def flush
      return if self.class.disabled?
      @mutex.synchronize do
        #return if @offset == @output.tell

      end

    def output=(o)
      return if self.class.disabled?
      @mutex.synchronize do
        if o.kind_of? String
          o = File.open(o, File::CREAT|File::TRUNC|File::RDWR, 0644)
        end
        @output = o
      end
    end
