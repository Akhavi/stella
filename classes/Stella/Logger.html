    def flush
      return if disabled?
      @mutex.synchronize do
        #return if @offset == @output.tell
        @buffer.seek @offset
        @output.print @buffer.read unless @buffer.eof?
        @offset = @buffer.tell
        @output.flush
      end
      true
    end

    def output=(o)
      @mutex.synchronize do
        if o.kind_of? String
          o = File.open(o, File::CREAT|File::TRUNC|File::RDWR, 0644)
        end
        @output = o
      end
    end
