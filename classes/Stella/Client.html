20:     def execute(usecase)
21:       http_client = generate_http_client
22:       container = Container.new(usecase)
23:       counter = 0
24:       usecase.requests.each do |req|
25:         counter += 1
26:         uri_obj = URI.parse(req.uri)
27:         params = prepare_params(usecase, req.params)
28:         uri = build_request_uri uri_obj, params, container
29:         raise NoHostDefined, uri_obj if uri.host.nil? || uri.host.empty?
30:         
31:         meth = req.http_method.to_s.downcase
32:         Stella.ld "#{meth}: " << "#{uri_obj.to_s} " << req.params.inspect
33:         
34:         begin
35:           update(:send_request, usecase, meth, uri, req, params, counter)
36:           container.response = http_client.send(meth, uri, params) # booya!
37:           update(:receive_response, usecase, meth, uri, req, params, container)
38:         rescue => ex
39:           update(:request_error, usecase, meth, uri, req, params, ex)
40:           next
41:         end
42:         
43:         ret = execute_response_handler container, req
44:         
45:         Drydock::Screen.flush
46:         
47:         if ret.kind_of?(ResponseModifier)
48:           case ret.class.to_s
49:           when "Stella::Client::Repeat"
50:             Stella.ld "REPETITION: #{counter} of #{ret.times+1}"
51:             redo if counter <= ret.times
52:           end
53:         end
54:         
55:         counter = 0 # reset
56:         run_sleeper(req.wait) if req.wait && !benchmark?
57:       end
58:     end
