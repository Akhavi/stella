 76:     def build_thread_package(plan, opts)
 77:       packages, pointer = Array.new(opts[:clients]), 0
 78:       plan.usecases.each_with_index do |usecase,i|
 79:         
 80:         count = case opts[:clients]
 81:         when 0..9
 82:           if (opts[:clients] % plan.usecases.size > 0) 
 83:             msg = "Client count does not evenly match usecase count"
 84:             raise Stella::Testplan::WackyRatio, msg
 85:           else
 86:             (opts[:clients] / plan.usecases.size)
 87:           end
 88:         else
 89:           (opts[:clients] * usecase.ratio).to_i
 90:         end
 91:         
 92:         Stella.ld "THREAD PACKAGE: #{usecase.desc} (#{pointer} + #{count})"
 93:         # Fill the thread_package with the contents of the block
 94:         packages.fill(pointer, count) do |index|
 95:           Stella.li2 "Creating client ##{index+1} "
 96:           client = Stella::Client.new opts[:hosts].first, index+1
 97:           client.add_observer(self)
 98:           client.enable_benchmark_mode if opts[:benchmark]
 99:           ThreadPackage.new(index+1, client, usecase.clone)
100:         end
101:         pointer += count
102:       end
103:       packages
104:     end
