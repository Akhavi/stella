75:     def calculate_usecase_clients(plan, opts)
76:       counts = { :total => 0 }
77:       plan.usecases.each_with_index do |usecase,i|
78:         count = case opts[:clients]
79:         when 0..9
80:           if (opts[:clients] % plan.usecases.size > 0) 
81:             msg = "Client count does not evenly match usecase count"
82:             raise Stella::Testplan::WackyRatio, msg
83:           else
84:             (opts[:clients] / plan.usecases.size)
85:           end
86:         else
87:           (opts[:clients] * usecase.ratio).to_i
88:         end
89:         counts[usecase.digest_cache] = count
90:         counts[:total] += count
91:       end
92:       counts
93:     end

113:     def execute_test_plan(packages, reps=1)
114:       #Thread.ify packages, :threads => packages.size do |package|
115:       threads = []
116:       packages.each { |package|
117:         threads << Thread.new do
118:           # This thread will stay on this one track. 
119:           Benelux.current_track package.client.gibbler
120:           Benelux.add_thread_tags :usecase => package.usecase.digest_cache
121:           (1..reps).to_a.each do |rep|
122:             Benelux.add_thread_tags :rep =>  rep
123:             Stella::Engine::Stress.rescue(package.client.digest_cache) {
124:               break if Stella.abort?
125:               print '.' if Stella.loglev == 2
126:               stats = package.client.execute package.usecase
127:             }
128:             Benelux.remove_thread_tags :rep
129:             sleep 0.001
130:           end
131:         
132:           Benelux.remove_thread_tags :usecase
133:         
134:         end
135:       }
136:       threads.each { |t| t.join } # wait
137:       Stella.li2 $/, $/
138:     end
