        def statuses_pretty
          pretty = ["Statuses"]
          if statuses.successful.size > 0
            pretty << '%20s: %s' % ['successful', statuses.successful.join(', ')] 
          end
          if statuses.nonsuccessful.size > 0
            pretty << '%20s: %s' % ['nonsuccessful', statuses.nonsuccessful.join(', ')] 
          end
          pretty.join $/
        end
