testplan_sync (plan)

Returns true if the testplan was created. Otherwise false if it already exists.

[show source] 
     # File lib/stella/service.rb, line 207
207:     def testplan_sync(plan)
208:       if testplan? plan.digest
209:         @tid = plan.digest
210:         return false 
211:       end
212:       Stella.stdout.info "Syncing Testplan #{plan.digest.short}"
213:       testplan_create plan.desc, :digest => plan.digest
214:       plan.usecases.each do |uc|
215:         Stella.stdout.info "Syncing Usecase #{uc.digest.short}"
216:         props = uc.to_hash
217:         props[:digest] ||= uc.digest
218:         usecase_create uc.desc, props
219:         uc.requests.each do |req|
220:           props = req.to_hash
221:           props[:digest] ||= req.digest
222:           props[:desc] = props.delete :description
223:           request_create props[:uri], props
224:         end
225:       end
226:       true
227:     end
