    def self.from_hash(hash={})
      me = super(hash)
      stats = {}
      me.stats.each_pair { |ucid,uchash| 
        stats[ucid] = {}
        uchash.each_pair { |reqid,reqhash|
          stats[ucid][reqid] = {}
          reqhash.each_pair { |event,stat|
            tmp = Benelux::Stats::Calculator.from_hash(stat)
            stats[ucid][reqid][event] = tmp
          }
        }
      }
      me.stats = stats
      me
    end
