    def start
      @dthread ||= Thread.new do
        prev_ptime = Time.now
        loop do
          @stopped = true and break if Stella.abort? || stop?
          if (Time.now - prev_ptime).to_i >= @freq
            @routine.call
            prev_ptime = Time.now
          end
          sleep @rest
        end
      end
      @dthread.abort_on_exception = true
    end

    def stop()  
      @force_stop = true  
      @dthread.join
      @routine.call
      @finally.call unless @finally.nil?
    end
